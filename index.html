  <script>

    let progress = 0;

    const progressBar = document.getElementById('progress-bar');

    const progressText = document.getElementById('progress-text');


    // Simulate the progress bar

    function simulateProgress() {

      if (progress < 100) {

        progress += Math.floor(Math.random() * 5) + 1; // Increment progress

        if (progress > 100) progress = 100; // Cap at 100%

        progressBar.style.width = progress + '%';

        progressText.textContent = progress + '%';

        setTimeout(simulateProgress, 200);

      } else {

        redirectToCheckout();

      }

    }


    // Redirect to the checkout page when complete

    function redirectToCheckout() {

      const checkoutUrl = "https://5fbqad-qz.myshopify.com/products/gift-service";

      window.location.href = checkoutUrl; // Redirect to Shopify product page

    }


    // Start progress simulation

    simulateProgress();

  </script>
